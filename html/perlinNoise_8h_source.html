53 float fade(float t)
54 {
55 return pow(t, 3.0f) * (t * (t * 6.0f - 15.0f) + 10.0f); // 6t^5 - 15t^4 + 10t^3
56 }
58 float Lerp(float a, float b, float t)
59 {
60 return (a * (1.0f - t)) + (b * t);
61 }

70 float Gradient(int hash, float x, float y, float z)
71 {
72 const int h = hash & 15;
73 const float u = h < 8 ? x : y, v = h < 4 ? y : h == 12 || h == 14 ? x : z;
74 return ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v);
75 }

83 float GetValue(float x, float y, float z)
84 {
85 // Unit coordinates in cube
86 const int X = int(floor(x)) & 255;
87 const int Y = int(floor(y)) & 255;
88 const int Z = int(floor(z)) & 255;
89
90 // Relative coordinates in cube
91 x = x - floor(x);
92 y = y - floor(y);
93 z = z - floor(z);
94
95 // Compute fading coefficients
96 const float u = fade(x);
97 const float v = fade(y);
98 const float w = fade(z);
99
100 // Hash cube coordinates
101 const int a = Perm[X] + Y;
102 const int aa = Perm[a] + Z;
103 const int ab = Perm[a + 1] + Z;
104 const int b = Perm[X + 1] + Y;
105 const int ba = Perm[b] + Z;
106 const int bb = Perm[b + 1] + Z;
107
108 // Interpolate results
109 const float l1 = Lerp(Gradient(Perm[aa], x, y, z), Gradient(Perm[ba], x - 1, y, z), u);
110 const float l2 = Lerp(Gradient(Perm[ab], x, y - 1, z), Gradient(Perm[bb], x - 1, y - 1, z), u);
111 const float l3 = Lerp(Gradient(Perm[aa + 1], x, y, z - 1), Gradient(Perm[ba + 1], x - 1, y, z - 1), u);
112 const float l4 = Lerp(Gradient(Perm[ab + 1], x, y - 1, z - 1), Gradient(Perm[bb + 1], x - 1, y - 1, z - 1), u);
113 const float l5 = Lerp(l1, l2, v);
114 const float l6 = Lerp(l3, l4, v);
115 return Lerp(l5, l6, w);
116 }
117};

152 float amp = amplitude, lac = lacunarity, pers = persistence, freq = frequency;
153 for(int i = 0; i < octaves; i++)

160 }
