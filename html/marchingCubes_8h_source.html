121 {

141 void calculateNormal(Mesh &mesh, int i, int j, int k)
142 {
143 Vector3 va = mesh.vertices[i];
144 Vector3 vb = mesh.vertices[j];
145 Vector3 vc = mesh.vertices[k];

151
152 mesh.normals[i] += normal;
153 mesh.normals[j] += normal;

159 {

161 unsigned int edgeIndices[12];
162 Vector3Unsigned* edgeIndexMap = new Vector3Unsigned[domain.size* domain.size * 2];
163 for(int i = 0; i <domain.size* domain.size * 2; i++ )
164 {
165 edgeIndexMap[i] = Vector3Unsigned({0,0,0});
166 }
167
168 for(int z = 0; z < domain.size - 1; z++)
169 {
170 for(int y = 0; y < domain.size - 1; y++)
171 {
172 for(int x = 0; x < domain.size - 1; x++)
173 {
174 unitCubeVertices[0] = domain.domain[x][y][z];
175 unitCubeVertices[1] = domain.domain[x+1][y][z];
176 unitCubeVertices[2] = domain.domain[x][y+1][z];
177 unitCubeVertices[3] = domain.domain[x+1][y+1][z];
178 unitCubeVertices[4] = domain.domain[x][y][z+1];
179 unitCubeVertices[5] = domain.domain[x+1][y][z+1];
180 unitCubeVertices[6] = domain.domain[x][y+1][z+1];
181 unitCubeVertices[7] = domain.domain[x+1][y+1][z+1];
182
183 int config_n = 0;
184 for(int i = 0; i < 8; i++)
185 {
186 config_n |= (unitCubeVertices[i] < isoLevel) << i;
187 }
188
189 if(config_n == 0 || config_n == 255)
190 continue;
191
192 if (y == 0 && z == 0)
193 addVertex(edgeIndexMap, mesh, unitCubeVertices[0], unitCubeVertices[1], 0, x, y, z, domain.size, isoLevel);
194 if (z == 0)
195 addVertex(edgeIndexMap, mesh, unitCubeVertices[2], unitCubeVertices[3], 0, x, y + 1, z, domain.size, isoLevel);
196 if (y == 0)
197 addVertex(edgeIndexMap, mesh, unitCubeVertices[4], unitCubeVertices[5], 0, x, y, z + 1, domain.size, isoLevel);
198 addVertex(edgeIndexMap, mesh, unitCubeVertices[6], unitCubeVertices[7], 0, x, y + 1, z + 1, domain.size, isoLevel);

202 if (z == 0)
203 addVertex(edgeIndexMap, mesh, unitCubeVertices[1], unitCubeVertices[3], 1,x + 1, y, z, domain.size, isoLevel);
204 if (x == 0)
205 addVertex(edgeIndexMap, mesh, unitCubeVertices[4], unitCubeVertices[6], 1, x, y, z + 1, domain.size, isoLevel);
206 addVertex(edgeIndexMap, mesh, unitCubeVertices[5], unitCubeVertices[7], 1, x + 1, y, z + 1, domain.size, isoLevel);
207
208 if (x == 0 && y == 0)
209 addVertex(edgeIndexMap, mesh, unitCubeVertices[0], unitCubeVertices[4], 2, x, y, z, domain.size, isoLevel);
210 if (y == 0)
211 addVertex(edgeIndexMap, mesh, unitCubeVertices[1], unitCubeVertices[5], 2, x + 1, y, z, domain.size, isoLevel);
212 if (x == 0)
213 addVertex(edgeIndexMap, mesh, unitCubeVertices[2], unitCubeVertices[6], 2, x, y + 1, z, domain.size, isoLevel);
214 addVertex(edgeIndexMap, mesh, unitCubeVertices[3], unitCubeVertices[7], 2, x + 1, y + 1, z, domain.size, isoLevel);
215
216 edgeIndices[0] = edgeIndexMap[index3DTo1D(x, y, z, domain.size)].x;
217 edgeIndices[1] = edgeIndexMap[index3DTo1D(x, y + 1, z, domain.size)].x;
218 edgeIndices[2] = edgeIndexMap[index3DTo1D(x, y, z + 1, domain.size)].x;
219 edgeIndices[3] = edgeIndexMap[index3DTo1D(x, y + 1, z + 1, domain.size)].x;
220 edgeIndices[4] = edgeIndexMap[index3DTo1D(x, y, z, domain.size)].y;
221 edgeIndices[5] = edgeIndexMap[index3DTo1D(x + 1, y, z, domain.size)].y;
222 edgeIndices[6] = edgeIndexMap[index3DTo1D(x, y, z + 1, domain.size)].y;
223 edgeIndices[7] = edgeIndexMap[index3DTo1D(x + 1, y, z + 1, domain.size)].y;
224 edgeIndices[8] = edgeIndexMap[index3DTo1D(x, y, z, domain.size)].z;
225 edgeIndices[9] = edgeIndexMap[index3DTo1D(x + 1, y, z, domain.size)].z;
226 edgeIndices[10] = edgeIndexMap[index3DTo1D(x, y + 1, z, domain.size)].z;
227 edgeIndices[11] = edgeIndexMap[index3DTo1D(x + 1, y + 1, z, domain.size)].z;
228
229 unsigned long long int config = mcConfig[config_n];
230 int nTriangles = config & 0xF;
231 int nIndices = nTriangles * 3;
232 int indicesBase = mesh.indices.size();
233 int offset = 4;
234
235 for(int i = 0; i < nIndices; i++)
236 {
237 int edge = (config >> offset) & 0xF;
238 mesh.indices.push_back(edgeIndices[edge]);
239 offset += 4;
240 }
241
242 for(int i = 0; i < nTriangles; i++)
243 {
244 calculateNormal(mesh, mesh.indices[indicesBase + i*3], mesh.indices[indicesBase + i*3 + 1], mesh.indices[indicesBase + i*3 + 2]);
245 }
246 }
247 }
248 }
249
250 for(int i = 0; i < mesh.normals.size(); i++)
251 {
252 mesh.normals[i].normalize();
253 }
254
